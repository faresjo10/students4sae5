# 1. Étape de Build (utilise une image Maven pour compiler)
FROM maven:3.9.5-eclipse-temurin-17-alpine AS build
WORKDIR /app
# Copie les fichiers de configuration Maven (pour le cache)
COPY pom.xml .
# Copie le code source
COPY src ./src
# Compile l'application et crée le JAR
RUN mvn clean package -DskipTests

# 2. Étape d'Exécution (utilise une image Java légère pour l'exécution)
FROM eclipse-temurin:17-jre-alpine
WORKDIR /app

# Récupère le JAR compilé depuis l'étape de 'build'
COPY --from=build /app/target/*.jar app.jar

# Expose le port de l'application (par défaut 8080 pour Spring Boot)
EXPOSE 8080

# Commande pour démarrer l'application (remplace SpringApp.jar par le nom réel si différent)
ENTRYPOINT ["java", "-jar", "app.jar"]
